--La somma dei Ram Sales USD divisa per il numero di mesi
member [Measures].[Avg Ram Sales USD] as
	round(sum([Measures].[Ram Sales USD])/count([Time].[Month].[Month]), 2)

([Geography].[Gerarchia].[Region], TOPCOUNT([Ram].[Ram Brand].[Ram Brand], 5, [Measures].[Avg Ram Sales USD]))  on rows
from [Group ID 3 DB]
