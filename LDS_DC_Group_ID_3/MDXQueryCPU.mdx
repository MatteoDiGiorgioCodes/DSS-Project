with

--La somma dei Cpu Sales USD divisa per il numero di mesi
member [Measures].[Avg Cpu Sales USD] as
	round(sum([Measures].[Cpu Sales USD])/count([Time].[Month].[Month]), 2)


select {[Measures].[Avg Cpu Sales USD] ,[Measures].[Cpu Sales USD]} on columns,
([Geography].[Gerarchia].[Region], TOPCOUNT([Cpu].[Cpu Brand].[Cpu Brand], 5, [Measures].[Avg Cpu Sales USD]))  on rows
from [Group ID 3 DB]
where {[Geography].[Continent].&[Europe]}
