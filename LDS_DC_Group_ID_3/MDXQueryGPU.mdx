with

--La somma dei Gpu Sales USD divisa per il numero di mesi
member [Measures].[Avg Gpu Sales USD] as
	round(sum([Measures].[Gpu Sales USD])/count([Time].[Month].[Month]), 2)


select {[Measures].[Avg Gpu Sales USD] ,[Measures].[Gpu Sales USD]} on columns,
([Geography].[Gerarchia].[Region], TOPCOUNT([Gpu].[Gpu Brand].[Gpu Brand], 5, [Measures].[Avg Gpu Sales USD]))  on rows
from [Group ID 3 DB]
where {[Geography].[Continent].&[Europe]}
