with

member [Measures].[Avg Ram Sales USD] as
	round(sum([Measures].[Ram Sales USD])/count([Time].[Month].[Month]), 2)

member [Measures].[Top Ram Brands] as
	GENERATE(
        TOPCOUNT([Ram].[Ram Brand].[Ram Brand], 5, [Measures].[Avg Ram Sales USD]),  -- Ottieni i top 5 brand
        [Ram].[Ram Brand].CurrentMember.Name,  -- Estrai il nome del brand
        ", "  -- Separa i brand con una virgola
    )



member [Measures].[Avg Cpu Sales USD] as
	round(sum([Measures].[Cpu Sales USD])/count([Time].[Month].[Month]), 2)

member [Measures].[Top Cpu Brands] as
	GENERATE(
        TOPCOUNT([Cpu].[Cpu Brand].[Cpu Brand], 5, [Measures].[Avg Cpu Sales USD]),  -- Ottieni i top 5 brand
        [Cpu].[Cpu Brand].CurrentMember.Name,  -- Estrai il nome del brand
        ", "  -- Separa i brand con una virgola
    )



member [Measures].[Avg Gpu Sales USD] as
	round(sum([Measures].[Gpu Sales USD])/count([Time].[Month].[Month]), 2)

member [Measures].[Top Gpu Brands] as
	GENERATE(
        TOPCOUNT([Gpu].[Gpu Brand].[Gpu Brand], 5, [Measures].[Avg Gpu Sales USD]),  -- Ottieni i top 5 brand
        [Gpu].[Gpu Brand].CurrentMember.Name,  -- Estrai il nome del brand
        ", "  -- Separa i brand con una virgola
    )

select {[Measures].[Top Ram Brands], [Measures].[Top Cpu Brands], [Measures].[Top Gpu Brands]} on columns,
([Geography].[Country].[Country], [Geography].[Region].[Region]) on rows
from [Group ID 3 DB]
where {[Geography].[Continent].&[Europe]}
